86 [[nodiscard]] static auto XM_CALLCONV compute_view(const transform& transform) noexcept -> DirectX::XMMATRIX {
87 const DirectX::XMVECTOR eye {DirectX::XMLoadFloat4(&transform.position) };
88 const DirectX::XMVECTOR focal {DirectX::XMVectorAdd(eye, transform.forward_vec()) };
89 return DirectX::XMMatrixLookAtLH(eye, focal, DirectX::XMVectorSet(.0F, 1.F, .0F, .0F));
90 }
91
92 [[nodiscard]] auto XM_CALLCONV compute_projection() const noexcept -> DirectX::XMMATRIX {
93 const float aspect = viewport.x / viewport.y;
94 return DirectX::XMMatrixPerspectiveFovLH(DirectX::XMConvertToRadians(this->fov), aspect, this->z_clip_near, this->z_clip_far);
