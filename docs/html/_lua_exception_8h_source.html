70 static bool areExceptionsEnabled(lua_State* L) noexcept
71 {
72 lua_pushlightuserdata(L, detail::getExceptionsKey());
73 lua_gettable(L, LUA_REGISTRYINDEX);
74
75 const bool enabled = lua_isboolean(L, -1) ? static_cast<bool>(lua_toboolean(L, -1)) : false;

86 static void enableExceptions(lua_State* L) noexcept
87 {
88 lua_pushlightuserdata(L, detail::getExceptionsKey());
89 lua_pushboolean(L, true);
90 lua_settable(L, LUA_REGISTRYINDEX);
91
92#if LUABRIDGE_HAS_EXCEPTIONS && LUABRIDGE_ON_LUAJIT
93 lua_pushlightuserdata(L, (void*)luajitWrapperCallback);
94 luaJIT_setmode(L, -1, LUAJIT_MODE_WRAPCFUNC | LUAJIT_MODE_ON);
95 lua_pop(L, 1);
96#endif
97
98#if LUABRIDGE_ON_LUAU
99 auto callbacks = lua_callbacks(L);
100 callbacks->panic = +[](lua_State* L, int) { panicHandlerCallback(L); };
101#else
102 lua_atpanic(L, panicHandlerCallback);
103#endif
104 }

124 void whatFromStack()
125 {
126 std::stringstream ss;
127
128 const char* errorText = nullptr;
129
130 if (lua_gettop(m_L) > 0)
131 {
132 errorText = lua_tostring(m_L, -1);
133 lua_pop(m_L, 1);
134 }
135
136 ss << (errorText ? errorText : "Unknown error") << " (code=" << m_code.message() << ")";
137
138 m_what = std::move(ss).str();
139 }

141 static int panicHandlerCallback(lua_State* L)

143#if LUABRIDGE_HAS_EXCEPTIONS

145#else
146 unused(L);

148 std::abort();
149#endif
150 }

152#if LUABRIDGE_HAS_EXCEPTIONS && LUABRIDGE_ON_LUAJIT
153 static int luajitWrapperCallback(lua_State* L, lua_CFunction f)
154 {
155 try
156 {
157 return f(L);
158 }
159 catch (const std::exception& e)
160 {
161 lua_pushstring(L, e.what());
162 return lua_error_x(L);
163 }
164 }
165#endif
166
167 lua_State* m_L = nullptr;
