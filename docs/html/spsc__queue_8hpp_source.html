148 template <typename T, typename Alloc>
149 spsc_queue<T, Alloc>::spsc_queue(const std::size_t capacity, const Alloc& allocator) noexcept {
150 static_assert(alignof(spsc_queue<T>) == k_hardware_destructive_interference_size, "Invalid alignment");
151 static_assert(sizeof(spsc_queue<T>) >= k_hardware_destructive_interference_size * 3, "Invalid size");
152 this->m_cap = capacity < 1 ? 0 : capacity;
153 this->m_alloc = allocator;
154 ++this->m_cap;
155 if (this->m_cap > eastl::numeric_limits<std::size_t>::max() - (k_padding << 1))
156 this->m_cap = eastl::numeric_limits<std::size_t>::max() - (k_padding << 1);
157 if constexpr(alloc_least<Alloc>::value) {
158 auto res = this->m_alloc.allocate_at_least(this->m_cap + (k_padding << 1));
159 this->m_slots = res.ptr;
160 this->m_cap = res.count - (k_padding << 1);
161 }
162 else this->m_slots = std::allocator_traits<Alloc>::allocate(this->m_alloc, this->m_cap + (k_padding << 1));
163 const bool is_valid = reinterpret_cast<const std::byte*>(&this->m_read_idx) - reinterpret_cast<const std::byte*>(&this->m_write_idx)
164 >= static_cast<std::ptrdiff_t>(k_hardware_destructive_interference_size);
165 passert(is_valid);

168 template <typename T, typename Alloc>
169 template <typename... Args> requires std::is_constructible_v<T, Args&&...>
170 auto spsc_queue<T, Alloc>::emplace(Args&&... args) noexcept(std::is_nothrow_constructible_v<T, Args &&...>) -> void
171 {
172 const std::size_t write_idx = this->m_write_idx.load(std::memory_order_relaxed);
173 std::size_t next_write = write_idx + 1;
174 if (next_write == this->m_cap) next_write = 0;
175 while (this->m_read_idx_cache == next_write)
176 this->m_read_idx_cache = this->m_read_idx.load(std::memory_order_acquire);
177 new (this->m_slots + write_idx + k_padding) T(std::forward<Args>(args)...);
178 this->m_write_idx.store(next_write, std::memory_order_release);
179 }

232 static_assert(std::is_nothrow_destructible_v<T>, "T must be nothrow destructible");
233 const std::size_t read_idx {this->m_read_idx.load(std::memory_order_relaxed) };
