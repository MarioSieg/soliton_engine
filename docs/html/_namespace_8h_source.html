39class Registrar
40{
41protected:
42 Registrar(lua_State* L)
43 : L(L)
44 , m_stackSize(0)
45 {
46 }
47
48 Registrar(lua_State* L, int skipStackPops)
49 : L(L)
50 , m_stackSize(0)
51 , m_skipStackPops(skipStackPops)
52 {
53 }
54
55 Registrar(const Registrar& rhs) = delete;
56
57 Registrar(Registrar&& rhs)
58 : L(rhs.L)
59 , m_stackSize(std::exchange(rhs.m_stackSize, 0))
60 , m_skipStackPops(std::exchange(rhs.m_skipStackPops, 0))
61 {
62 }
63
64 Registrar& operator=(const Registrar& rhs) = delete;
65
66 Registrar& operator=(Registrar&& rhs)
67 {
68 m_stackSize = std::exchange(rhs.m_stackSize, 0);
69 m_skipStackPops = std::exchange(rhs.m_skipStackPops, 0);
70
71 return *this;
72 }
73
74 ~Registrar()
75 {
76 const int popsCount = m_stackSize - m_skipStackPops;
77 if (popsCount > 0)
78 {
79 LUABRIDGE_ASSERT(popsCount <= lua_gettop(L));
80
81 lua_pop(L, popsCount);
82 }
83 }
84
85 void assertIsActive() const
86 {
87 if (m_stackSize == 0)
88 {
89 throw_or_assert<std::logic_error>("Unable to continue registration");
90 }
91 }
92
93 lua_State* const L = nullptr;
94 int m_stackSize = 0;
95 int m_skipStackPops = 0;
96};

165 void createConstTable(const char* name, bool trueConst, Options options)
166 {
167 LUABRIDGE_ASSERT(name != nullptr);
168
169 std::string type_name = std::string(trueConst ? "const " : "") + name;
170
171 // Stack: namespace table (ns)
172
173 lua_newtable(L); // Stack: ns, const table (co)
174 lua_pushvalue(L, -1); // Stack: ns, co, co
175 lua_setmetatable(L, -2); // co.__metatable = co. Stack: ns, co
176
177 pushunsigned(L, options.toUnderlying()); // Stack: ns, co, options
178 lua_rawsetp(L, -2, detail::getClassOptionsKey()); // co [classOptionsKey] = options. Stack: ns, co
179
180 lua_pushstring(L, type_name.c_str()); // Stack: ns, co, name
181 lua_rawsetp(L, -2, detail::getTypeKey()); // co [typeKey] = name. Stack: ns, co
182
183 lua_pushcfunction_x(L, &detail::index_metamethod<true>); // Stack: ns, co, im
184 rawsetfield(L, -2, "__index"); // Stack: ns, co
185
186 lua_pushcfunction_x(L, &detail::newindex_metamethod<true>); // Stack: ns, co, nim
187 rawsetfield(L, -2, "__newindex"); // Stack: ns, co
188
189 lua_newtable(L); // Stack: ns, co, tb
190 lua_rawsetp(L, -2, detail::getPropgetKey()); // Stack: ns, co
191
192 if (! options.test(visibleMetatables))
193 {
194 lua_pushboolean(L, 0);
195 rawsetfield(L, -2, "__metatable");
196 }

242 lua_rawsetp(L, -2, detail::getClassOptionsKey()); // st [classOptionsKey] = options. Stack: ns, co, cl, st, mt
243
244 lua_pushcfunction_x(L, &detail::index_metamethod<false>);
245 rawsetfield(L, -2, "__index");
246
247 lua_pushcfunction_x(L, &detail::newindex_metamethod<false>);
248 rawsetfield(L, -2, "__newindex");
249

252

254 lua_rawsetp(L, -2, detail::getPropsetKey()); // st [propsetKey] = pg. Stack: ns, co, cl, st

260 {
261 lua_pushboolean(L, 0);
262 rawsetfield(L, -2, "__metatable");
263 }

324 ++m_stackSize;
325#if !defined(LUABRIDGE_ON_LUAU)
326 lua_pushcfunction_x(L, &detail::gc_metamethod<T>); // Stack: ns, co, cl, function

334 ++m_stackSize;

344 lua_pushvalue(L, -2); // Stack: ns, co, cl, st, cl
345 lua_rawsetp(L, LUA_REGISTRYINDEX, detail::getClassRegistryKey<T>()); // Stack: ns, co, cl, st

352 ++m_stackSize;
353
354 lua_getmetatable(L, -1); // Stack: ns, vst, st
355 lua_insert(L, -2); // Stack: ns, st, vst
356 lua_pop(L, 1); // Stack: ns, st
357
358 // Map T back from its stored tables
359 lua_rawgetp(L, LUA_REGISTRYINDEX, detail::getConstRegistryKey<T>()); // Stack: ns, st, co
360 lua_insert(L, -2); // Stack: ns, co, st
361 ++m_stackSize;
362
363 lua_rawgetp(L, LUA_REGISTRYINDEX, detail::getClassRegistryKey<T>()); // Stack: ns, co, st, cl
364 lua_insert(L, -2); // Stack: ns, co, cl, st
365 ++m_stackSize;
366 }

633 template <class... Functions>
634 auto addStaticFunction(const char* name, Functions... functions)
635 -> std::enable_if_t<(detail::is_callable_v<Functions> && ...) && (sizeof...(Functions) > 0), Class<T>&>
636 {
637 LUABRIDGE_ASSERT(name != nullptr);
638 assertStackState(); // Stack: const table (co), class table (cl), static table (st)
639
640 if constexpr (sizeof...(Functions) == 1)
641 {
642 ([&]
643 {
644 detail::push_function(L, std::move(functions));
645
646 } (), ...);
647 }
648 else
649 {
650 // create new closure of try_overloads with new table
651 lua_createtable(L, static_cast<int>(sizeof...(Functions)), 0); // reserve space for N overloads
652
653 int idx = 1;
654
655 ([&]
656 {
657 lua_createtable(L, 2, 0); // reserve space for: function, arity
658 lua_pushinteger(L, 1);
659 if constexpr (detail::is_any_cfunction_pointer_v<Functions>)
660 lua_pushinteger(L, -1);
661 else
662 lua_pushinteger(L, static_cast<int>(detail::function_arity_excluding_v<Functions, lua_State*>));
663 lua_settable(L, -3);
664 lua_pushinteger(L, 2);
665 detail::push_function(L, std::move(functions));
666 lua_settable(L, -3);
667
668 lua_rawseti(L, -2, idx);
669 ++idx;
670
671 } (), ...);
672
673 lua_pushcclosure_x(L, &detail::try_overload_functions<false>, 1);
674 }
675
676 rawsetfield(L, -2, name);
677
678 return *this;
679 }
