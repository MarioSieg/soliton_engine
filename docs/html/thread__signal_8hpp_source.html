33 inline auto thread_sig::trigger(const bool notify_all, const std::int32_t value) -> void {
34 // The thread that intends to modify the variable has to
35 // -> acquire a std::mutex (typically via std::lock_guard)
36 // -> perform the modification while the lock is held
37 // -> execute notify_one or notify_all on the std::condition_variable (the lock does not need to be held for notification)
38 passert(value != 0);
39 {
40 std::unique_lock lock {m_mtx};
41 passert(!m_signaled.load(std::memory_order_seq_cst) && !m_threads_awaken.load(std::memory_order_seq_cst));
42 m_signaled.store(value, std::memory_order_seq_cst);
43 }
44 // Unlocking is done before notifying, to avoid waking up the waiting
45 // thread only to block again (see notify_one for details)
46 if (notify_all){
47 m_cond.notify_all();
48 } else {
49 m_cond.notify_one();
50 }
51 }
52
53 // WARNING!
54 // If multiple threads are waiting for a signal in an infinite loop,
55 // autoresetting the signal does not guarantee that one thread cannot
56 // go through the loop twice. In this case, every thread must wait for its
57 // own auto-reset signal or the threads must be blocked by another signal
58 inline auto thread_sig::wait(const bool auto_reset, const std::int32_t num_threads_waiting) -> std::int32_t {
59 // Any thread that intends to wait on std::condition_variable has to
60 // * acquire a std::unique_lock<std::mutex>, on the SAME MUTEX as used to protect the shared variable
61 // * execute wait, wait_for, or wait_until. The wait operations atomically release the mutex
62 // and suspend the execution of the thread.
63 // * When the condition variable is notified, a timeout expires, or a spurious wakeup occurs,
64 // the thread is awakened, and the mutex is atomically reacquired:
65 // - The thread should then check the condition and resume waiting if the wake-up was spurious.
66 std::unique_lock lock {m_mtx};
67 if (!m_signaled.load(std::memory_order_seq_cst)) {
68 m_cond.wait(lock, [this]() noexcept -> bool { return m_signaled.load(std::memory_order_seq_cst); });
69 }
70 const std::int32_t signaled_value = m_signaled.load(std::memory_order_seq_cst);
71 const std::int32_t num_thread_awaken = 1 + m_threads_awaken.fetch_add(1, std::memory_order_seq_cst); // fetch_add returns the original value immediately preceding the addition.
72 if (auto_reset) {
73 passert(num_threads_waiting != 0);
74 if (num_thread_awaken == num_threads_waiting) {
75 m_signaled.store(0, std::memory_order_seq_cst);
76 m_threads_awaken.store(0, std::memory_order_seq_cst);
77 }
78 }
79 return signaled_value;
80 }
