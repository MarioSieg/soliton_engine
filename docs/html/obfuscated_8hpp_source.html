19 // This class obfuscates the contained data in memory by XORing it with a magic number
20 // and the address of the data itself. Avoid bitwise copying of this type!
21 // The contained type in sysObfuscated does not need to be a POD type, but note:
22 // 1. External code accessing the object through its 'this' pointer will get obfuscated data, likely causing a crash.
23 // 2. The copy constructor and assignment operator may be called in non-standard ways, potentially with a misaligned object.
24 //
25 // Template arguments:
26 // T: The type to be obfuscated (must be a multiple of 32 bits).
27 // Mutate: If true, the class size doubles, and the memory layout changes with each read, making it harder to detect data changes with breakpoints.
28 // This has a performance impact compared to non-mutated types.
29 template <typename T, bool Mutate=true> requires ((sizeof(T) & 3) == 0)
30 class obfuscated final : public no_copy, public no_move {
31 LU_FORCEINLINE obfuscated() noexcept {
32 init();
33 }
34 ~obfuscated() = default;

36 LU_FORCEINLINE auto get() const noexcept -> T {
37 const std::uint32_t xor_v = m_xor ^ (std::uint32_t)(size_t)this;
38 eastl::array<std::uint32_t, sizeof(T)/sizeof(std::uint32_t)> ret;
39 auto* src = const_cast<std::uint32_t*>(&m_data[0]);
40 auto* const dest = ret.data();
41 for (std::size_t i = 0; i < (sizeof(T)>>2); ++i) {
42 if constexpr (Mutate) {
43 const std::uint32_t x = *src & m_mutate;
44 const std::uint32_t y = src[sizeof(T) >> 2] & (~m_mutate);
45 const std::uint32_t entropy_x = ((*src&(~m_mutate))<<16) | ((*src&(~m_mutate))>>16);
46 const std::uint32_t entropy_y = ((src[sizeof(T)>>2]&m_mutate)<<16) | ((src[sizeof(T)>>2]&m_mutate)>>16);
47 *src = (*src & m_mutate) | entropy_x;
48 src[sizeof(T)>>2] = (src[sizeof(T)>>2]&(~m_mutate))|entropy_y;
49 *dest++ = x|y;
50 ++src;
51 }
52 else
53 {
54 *dest++ = *src++ ^ xor_v;
55 }
56 }
57 if constexpr (Mutate) {
58 const_cast<obfuscated<T, Mutate>*>(this)->set(eastl::bit_cast<T>(ret));
59 }
60 return eastl::bit_cast<T>(ret);
61 }

63 LU_FORCEINLINE auto set(const T& value) noexcept -> void {
64 init();
65 const std::uint32_t xor_v = m_xor ^ static_cast<std::uint32_t>(eastl::bit_cast<std::uintptr_t>(this));
66 auto* src = reinterpret_cast<const std::uint32_t*>(&value);
67 auto* const dest = m_data.data();
68 for (std::size_t i = 0; i < (sizeof(T)>>2); ++i) {
69 if constexpr (Mutate) {
70 const std::uint32_t a = *src & m_mutate;
71 const std::uint32_t b = *src & ~m_mutate;
72 ++src;
73 *dest = a;
74 dest[sizeof(T)>>2] = b;
75 ++dest;
76 } else {
77 *dest++ = *src++ ^ xor_v;
78 }
79 }
80 }
