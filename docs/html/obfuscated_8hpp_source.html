19 // This class obfuscates the contained data in memory by XORing it with a magic number
20 // and the address of the data itself. Avoid bitwise copying of this type!
21 // The contained type in sysObfuscated does not need to be a POD type, but note:
22 // 1. External code accessing the object through its 'this' pointer will get obfuscated data, likely causing a crash.
23 // 2. The copy constructor and assignment operator may be called in non-standard ways, potentially with a misaligned object.
24 //
25 // Template arguments:
26 // T: The type to be obfuscated (must be a multiple of 32 bits).
27 // Mutate: If true, the class size doubles, and the memory layout changes with each read, making it harder to detect data changes with breakpoints.
28 // This has a performance impact compared to non-mutated types.
29 template <typename T, bool Mutate=true>
30 class obfuscated final : public no_copy, public no_move {
31 static_assert((sizeof(T) & 3) == 0, "obfuscated<T> requires T to be a multiple of 32 bits");
32
33 LU_FORCEINLINE obfuscated() noexcept {
34 init();
35 }
36 ~obfuscated() = default;
37
38 LU_FORCEINLINE auto get() const noexcept -> T {
39 const std::uint32_t xor_v = m_xor ^ (std::uint32_t)(size_t)this;
40 std::array<std::uint32_t, sizeof(T)/sizeof(std::uint32_t)> ret;
41 auto* src = const_cast<std::uint32_t*>(&m_data[0]);
42 auto* const dest = ret.data();
43 for (std::size_t i = 0; i < (sizeof(T)>>2); ++i) {
44 if constexpr (Mutate) {
45 const std::uint32_t x = *src & m_mutate;
46 const std::uint32_t y = src[sizeof(T) >> 2] & (~m_mutate);
47 const std::uint32_t entropy_x = ((*src&(~m_mutate))<<16) | ((*src&(~m_mutate))>>16);
48 const std::uint32_t entropy_y = ((src[sizeof(T)>>2]&m_mutate)<<16) | ((src[sizeof(T)>>2]&m_mutate)>>16);
49 *src = (*src & m_mutate) | entropy_x;
50 src[sizeof(T)>>2] = (src[sizeof(T)>>2]&(~m_mutate))|entropy_y;
51 *dest++ = x|y;
52 ++src;
53 }
54 else
55 {
56 *dest++ = *src++ ^ xor_v;
57 }
58 }
59 if constexpr (Mutate) {
60 const_cast<obfuscated<T, Mutate>*>(this)->set(std::bit_cast<T>(ret));
61 }
62 return std::bit_cast<T>(ret);
63 }

65 LU_FORCEINLINE auto set(const T& value) noexcept -> void {
66 init();
67 const std::uint32_t xor_v = m_xor ^ static_cast<std::uint32_t>(std::bit_cast<std::uintptr_t>(this));
68 auto* src = reinterpret_cast<const std::uint32_t*>(&value);
69 auto* const dest = m_data.data();
70 for (std::size_t i = 0; i < (sizeof(T)>>2); ++i) {
71 if constexpr (Mutate) {
72 const std::uint32_t a = *src & m_mutate;
73 const std::uint32_t b = *src & ~m_mutate;
74 ++src;
75 *dest = a;
76 dest[sizeof(T)>>2] = b;
77 ++dest;
78 } else {
79 *dest++ = *src++ ^ xor_v;
80 }
81 }
82 }
