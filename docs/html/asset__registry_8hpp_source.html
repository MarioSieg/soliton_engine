10 enum class asset_ref : std::int32_t {}; // Asset reference type to access asset instances by index (used from Lua)
11 static constexpr asset_ref asset_ref_invalid = static_cast<asset_ref>(std::numeric_limits<std::underlying_type_t<asset_ref>>::max());
12 static_assert(0x7fffffff == static_cast<std::size_t>(asset_ref_invalid)); // Invalid asset ref constant also used in Lua

14 // Maximum number of valid asset references

18 class asset_registry final {

25 [[nodiscard]] auto insert(eastl::unique_ptr<T>&& asset = nullptr, const eastl::string_view path = {}) -> eastl::pair<asset_ref, T*> {
26 if constexpr (std::is_constructible_v<T>) {
27 if (!asset) {
28 asset = eastl::make_unique<T>();
29 }
30 }
31 panic_assert(asset != nullptr);
32 if (!path.empty()) { // Check if asset with path already exists
33 const auto it = m_path_cache.find(path);
34 if (it != m_path_cache.end()) { // Return existing asset reference
35 T* const loaded = (*this)[it->second];
36 panic_assert(loaded != nullptr);
37 return eastl::make_pair(it->second, loaded);
38 }
39 }
40 const std::size_t idx = m_loaded.size();
41 if (idx > static_cast<std::size_t>(asset_ref_max)) [[unlikely]] {
42 log_error("Asset asset_ref numeric capacity exhausted, cannot insert asset: {}", path);
43 return eastl::make_pair(asset_ref_invalid, nullptr);
44 }
45 const auto ref = static_cast<asset_ref>(idx);
46 T* const ptr = &*asset;
47 m_loaded.emplace_back(eastl::move(asset));
48 if (!path.empty()) {
49 m_path_cache.emplace(path, ref);
50 }
51 return eastl::make_pair(ref, ptr);
52 }
53
54 [[nodiscard]] auto load(eastl::string&& path) -> asset_ref {
55 if (path.empty()) {
56 log_error("Cannot load asset with empty path");
57 return asset_ref_invalid;
58 }
59 const auto it = m_path_cache.find(path);
60 if (it != m_path_cache.end()) {
61 return it->second;
62 }
63 eastl::string_view path_view { path };
64 auto asset = eastl::make_unique<T>(eastl::move(path));
65 if (!asset) {
66 log_error("Failed to load asset from path: {}", path_view);
67 return asset_ref_invalid;
68 }
69 return insert(eastl::move(asset), path_view).first;
70 }
