86 static void enableExceptions(lua_State* L) noexcept
87 {

94 luaJIT_setmode(L, -1, LUAJIT_MODE_WRAPCFUNC | LUAJIT_MODE_ON);
95 lua_pop(L, 1);
96#endif
97
98#if LUABRIDGE_ON_LUAU
99 auto callbacks = lua_callbacks(L);

124 void whatFromStack()
125 {
126 std::stringstream ss;
127
128 const char* errorText = nullptr;
129
130 if (lua_gettop(m_L) > 0)
131 {
132 errorText = lua_tostring(m_L, -1);
133 lua_pop(m_L, 1);
134 }
135
136 ss << (errorText ? errorText : "Unknown error") << " (code=" << m_code.message() << ")";
137
138 m_what = std::move(ss).str();
139 }

154 {
155 try

159 catch (const std::exception& e)
