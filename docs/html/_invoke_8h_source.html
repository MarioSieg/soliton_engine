110 static LuaResult errorFromStack(lua_State* L, std::error_code ec)
111 {
112 auto errorString = lua_tostring(L, -1);
113 lua_pop(L, 1);
114
115 return LuaResult(L, ec, errorString ? errorString : ec.message());

118 static LuaResult valuesFromStack(lua_State* L, int stackTop)
119 {
120 std::vector<LuaRef> values;
121
122 const int numReturnedValues = lua_gettop(L) - stackTop;
123 if (numReturnedValues > 0)
124 {
125 values.reserve(numReturnedValues);
126
127 for (int index = numReturnedValues; index > 0; --index)
128 values.emplace_back(LuaRef::fromStack(L, -index));
129
130 lua_pop(L, numReturnedValues);
131 }
132
133 return LuaResult(L, std::move(values));
134 }

171template <class... Args>
172LuaResult call(const LuaRef& object, Args&&... args)
173{
174 lua_State* L = object.state();
175 const int stackTop = lua_gettop(L);
176
177 object.push();
178
179 {
180 const auto [result, index] = detail::push_arguments(L, std::forward_as_tuple(args...));
181 if (! result)
182 {
183 lua_pop(L, static_cast<int>(index) + 1);
184 return LuaResult(L, result, result.message());
185 }
186 }
187
188 const int code = lua_pcall(L, sizeof...(Args), LUA_MULTRET, 0);
189 if (code != LUABRIDGE_LUA_OK)
190 {
191 auto ec = makeErrorCode(ErrorCode::LuaFunctionCallFailed);
192
193#if LUABRIDGE_HAS_EXCEPTIONS
194 if (LuaException::areExceptionsEnabled(L))
195 LuaException::raise(L, ec);
196#endif
197
198 return LuaResult::errorFromStack(L, ec);
199 }
200
201 return LuaResult::valuesFromStack(L, stackTop);
202}

208inline int pcall(lua_State* L, int nargs = 0, int nresults = 0, int msgh = 0)
209{
210 const int code = lua_pcall(L, nargs, nresults, msgh);
211
212#if LUABRIDGE_HAS_EXCEPTIONS
213 if (code != LUABRIDGE_LUA_OK && LuaException::areExceptionsEnabled(L))
214 LuaException::raise(L, makeErrorCode(ErrorCode::LuaFunctionCallFailed));
215#endif
216
217 return code;
218}
