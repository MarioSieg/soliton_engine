52 template <typename Scalar, const std::size_t Size> requires std::is_arithmetic_v<Scalar>
53 auto fixed_ring_scalar_buffer<Scalar, Size>::submit(const Scalar value) noexcept -> void {
54 this->m_buf[this->m_offs] = value;
55 this->m_offs = (this->m_offs + 1) % Size;
56 Scalar min = eastl::numeric_limits<Scalar>::max();
57 Scalar max = eastl::numeric_limits<Scalar>::min();
58 Scalar avg {};
59 for (const Scalar* __restrict__ i { this->m_buf.begin() }, *const __restrict__ e = i + this->m_buf.size(); i < e; ++i) {
60 min = std::min(min, *i);
61 max = std::max(max, *i);
62 avg += *i;
63 }
64 this->m_min = min;
65 this->m_max = max;
66 this->m_avg = avg / static_cast<Scalar>(Size);
67 }
