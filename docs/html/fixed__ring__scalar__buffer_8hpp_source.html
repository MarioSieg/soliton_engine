53 template <typename Scalar, const std::size_t Size> requires std::is_arithmetic_v<Scalar>
54 auto fixed_ring_scalar_buffer<Scalar, Size>::submit(const Scalar value) noexcept -> void {
55 this->m_buf[this->m_offs] = value;
56 this->m_offs = (this->m_offs + 1) % Size;
57 Scalar min = std::numeric_limits<Scalar>::max();
58 Scalar max = std::numeric_limits<Scalar>::min();
59 Scalar avg {};
60 for (const Scalar* __restrict__ i { this->m_buf.begin() }, *const __restrict__ e = i + this->m_buf.size(); i < e; ++i)
61 {
62 min = std::min(min, *i);
63 max = std::max(max, *i);
64 avg += *i;
65 }
66 this->m_min = min;
67 this->m_max = max;
68 this->m_avg = avg / static_cast<Scalar>(Size);
69 }
