195inline int lua_compare(lua_State* L, int idx1, int idx2, int op)
196{
197 switch (op)
198 {
199 case LUA_OPEQ:
200 return lua_equal(L, idx1, idx2);
201
202 case LUA_OPLT:
203 return lua_lessthan(L, idx1, idx2);
204
205 case LUA_OPLE:
206 return lua_equal(L, idx1, idx2) || lua_lessthan(L, idx1, idx2);
207
208 default:
209 return 0;
210 }
211}

349{
350#if LUA_VERSION_NUM < 502
351 lua_getglobal(threadL, main_thread_name);
352 if (lua_isthread(threadL, -1))
353 {
354 auto L = lua_tothread(threadL, -1);
355 lua_pop(threadL, 1);

441template <class T>
442[[nodiscard]] T* align(void* ptr) noexcept
443{
444 const auto address = reinterpret_cast<size_t>(ptr);
445
446 const auto offset = address % alignof(T);
447 const auto aligned_address = (offset == 0) ? address : (address + alignof(T) - offset);
448
449 return reinterpret_cast<T*>(aligned_address);
450}

519inline int raise_lua_error(lua_State* L, const char* fmt, ...)
520{
521 va_list argp;
522 va_start(argp, fmt);
523 lua_pushvfstring(L, fmt, argp);
524 va_end(argp);
525
526 const char* message = lua_tostring(L, -1);
527 if (message != nullptr)
528 {
529 if (auto str = std::string_view(message); !str.empty() && str[0] == '[')
530 return lua_error_x(L);
531 }
532
533 bool pushed_error = false;
534 for (int level = 1; level <= 2; ++level)
535 {
536 lua_Debug ar;
537
538#if LUABRIDGE_ON_LUAU
539 if (lua_getinfo(L, level, "sl", &ar) == 0)
540 continue;
541#else
542 if (lua_getstack(L, level, &ar) == 0 || lua_getinfo(L, "Sl", &ar) == 0)
543 continue;
544#endif
545
546 if (ar.currentline <= 0)
547 continue;
548
549 lua_pushfstring(L, "%s:%d: ", ar.short_src, ar.currentline);
550 pushed_error = true;
551
552 break;
553 }
554
555 if (! pushed_error)
556 lua_pushliteral(L, "");
557
558 lua_pushvalue(L, -2);
559 lua_remove(L, -3);
560 lua_concat(L, 2);
561
562 return lua_error_x(L);
563}

568template <class U = lua_Integer, class T>
569constexpr bool is_integral_representable_by(T value)
570{
571 constexpr bool same_signedness = (std::is_unsigned_v<T> && std::is_unsigned_v<U>)
572 || (!std::is_unsigned_v<T> && !std::is_unsigned_v<U>);
573
574 if constexpr (sizeof(T) == sizeof(U))
575 {
576 if constexpr (same_signedness)
577 return true;
578
579 if constexpr (std::is_unsigned_v<T>)
580 return value <= static_cast<T>((std::numeric_limits<U>::max)());
581
582 return value >= static_cast<T>((std::numeric_limits<U>::min)())
583 && static_cast<U>(value) <= (std::numeric_limits<U>::max)();
584 }
585
586 if constexpr (sizeof(T) < sizeof(U))
587 {
588 return static_cast<U>(value) >= (std::numeric_limits<U>::min)()
589 && static_cast<U>(value) <= (std::numeric_limits<U>::max)();
590 }
591
592 if constexpr (std::is_unsigned_v<T>)
593 return value <= static_cast<T>((std::numeric_limits<U>::max)());
594
595 return value >= static_cast<T>((std::numeric_limits<U>::min)())
596 && value <= static_cast<T>((std::numeric_limits<U>::max)());
597}

612template <class U = lua_Number, class T>
613constexpr bool is_floating_point_representable_by(T value)
614{
615 if constexpr (sizeof(T) == sizeof(U))
616 return true;
617
618 if constexpr (sizeof(T) < sizeof(U))
619 return static_cast<U>(value) >= -(std::numeric_limits<U>::max)()
620 && static_cast<U>(value) <= (std::numeric_limits<U>::max)();
621
622 return value >= static_cast<T>(-(std::numeric_limits<U>::max)())
623 && value <= static_cast<T>((std::numeric_limits<U>::max)());
624}
