49 [[nodiscard]] static TypeResult<Type> get(lua_State* L, int index)
50 {
51 if (!lua_istable(L, index))
52 return makeErrorCode(ErrorCode::InvalidTypeCast);
53
54 const StackRestore stackRestore(L);
55
56 Type vector;
57 vector.reserve(static_cast<std::size_t>(get_length(L, index)));
58
59 int absIndex = lua_absindex(L, index);
60 lua_pushnil(L);
61
62 while (lua_next(L, absIndex) != 0)
63 {
64 auto item = Stack<T>::get(L, -1);
65 if (! item)
66 return makeErrorCode(ErrorCode::InvalidTypeCast);
67
68 vector.emplace_back(*item);
69 lua_pop(L, 1);
70 }

73 }
