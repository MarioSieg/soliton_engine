34 directory.Assign(path, pos);

36

38 {
39 void* watcher = FSW::Create(directory.Str(), false);
40 FSW::Changed(watcher) += MakeDelegate(this, &ProviderFileWatcher::ChangeNotification);

42 mFileSystemWatchers.PushBack(watcher);
43 }
44 }
45 }
