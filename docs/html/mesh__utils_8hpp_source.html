24 switch (pOrigin) {
25 case aiOrigin_SET:
26 return m_stream->set_pos(static_cast<std::streamsize>(pOffset), std::ios::beg) ? aiReturn_SUCCESS : aiReturn_FAILURE;
27 case aiOrigin_CUR:
28 return m_stream->set_pos(static_cast<std::streamsize>(pOffset), std::ios::cur) ? aiReturn_SUCCESS : aiReturn_FAILURE;
29 case aiOrigin_END:
30 return m_stream->set_pos(static_cast<std::streamsize>(pOffset), std::ios::end) ? aiReturn_SUCCESS : aiReturn_FAILURE;
31 default: panic("Unknown origin");
32 }
33 }
