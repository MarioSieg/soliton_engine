46 template <typename T> requires is_con_var_type<T>
47 class convar final {
48 public:
49 convar(
50 std::string&& name,
51 std::optional<std::variant<T, std::function<auto() -> T>>>&& fallback,
52 std::underlying_type_t<convar_flags::$> flags,
53 const T min = {},
54 const T max = {}
55 ) noexcept : m_name{std::move(name)}, m_fallback{std::move(fallback)}, m_flags{flags}, m_min{min}, m_max{max} {}

57 [[nodiscard]] auto operator ()() -> T {
58 if (m_is_locked) [[unlikely]] {
59 log_error("Failed to get CONVAR '{}': Variable write is locked", full_name());
60 return fallback();
61 }
62 register_var();
63 if (!m_ref) [[unlikely]] {
64 log_error("Failed to get CONVAR '{}': Engine config key not valid", full_name());
65 return fallback();
66 }
67 ++m_gets;
68 const T rv = m_ref->cast<T>().valueOr(fallback());
69 if constexpr (!std::is_same_v<T, bool> && std::is_integral_v<T> || std::is_floating_point_v<T>) {
70 if (~(m_flags & convar_flags::no_clamp)) {
71 return std::clamp(rv, m_min, m_max);
72 }
73 }
74 return rv;
75 }

77 [[nodiscard]] auto full_name() const -> std::string_view { return m_name; }
78
79 private:
80 auto register_var() {
81 if (m_ref) return;
82 passert(detail::tid == std::this_thread::get_id() && "Convars must be registered on the main thread");
83 std::string s_fallback {"None"};
84 if constexpr (std::is_same_v<T, std::string> | std::is_same_v<T, bool>) {
85 if (m_fallback) {
86 if constexpr (std::is_same_v<T, std::string>) s_fallback = fallback();
87 else s_fallback = fallback() ? "true" : "false";
88 }
89 log_info("Registering CONVAR #{} [{} : {}] | Flags: {:#x}, Fallback: {}", ++detail::s_convar_i, full_name(), type_name(), m_flags, s_fallback);
90 } else {
91 if (m_fallback) {
92 s_fallback = std::to_string(fallback());
93 }
94 bool has_range = false;
95 if constexpr(std::is_floating_point_v<T>) {
96 has_range = std::abs(m_min) > std::numeric_limits<T>::epsilon() || std::abs(m_max) > std::numeric_limits<T>::epsilon();
97 } else {
98 has_range = m_min != std::numeric_limits<T>::min() || m_max != std::numeric_limits<T>::max();
99 }
100 std::string range {};
101 if (has_range) {
102 range = fmt::format(", Range: [{}, {}]", m_min, m_max);
103 }
104 log_info("Registering CONVAR #{} [{} : {}] | Flags: {:#x}, Fallback: {}{}", ++detail::s_convar_i, full_name(), type_name(), m_flags, s_fallback, range);
105 }
106 if constexpr (!std::is_same_v<T, bool> && std::is_integral_v<T> || std::is_floating_point_v<T>) {
107 if (m_min > m_max) [[unlikely]] {
108 panic("Failed to create CONVAR '{}': The min value is greater than the max value", full_name());
109 return;
110 }
111 if constexpr (std::is_floating_point_v<T>) {
112 if (std::abs(m_min) < std::numeric_limits<T>::epsilon())
113 m_min = std::numeric_limits<T>::min();
114 if (std::abs(m_max) < std::numeric_limits<T>::epsilon())
115 m_max = std::numeric_limits<T>::max();
116 } else {
117 if (m_min == 0) m_min = std::numeric_limits<T>::min();
118 if (m_max == 0) m_max = std::numeric_limits<T>::max();
119 }
120 } else if constexpr (std::is_same_v<T, std::string>) {
121 if (!m_min.empty() || !m_max.empty()) [[unlikely]] {
122 panic("Failed to create CONVAR '{}', string/bool variables cannot have min/max clamps", full_name());
123 return;
124 }
125 }
126 if (m_name.empty()) [[unlikely]] {
127 panic("Failed to create CONVAR: Empty name");
128 return;
129 }
130 std::stack<luabridge::LuaRef> refs {};
131 const auto* const root = scripting_subsystem::cfg();
132 if (!root || !root->isTable()) [[unlikely]] {
133 log_error("Failed to register CONVAR '{}': Engine config root table is not initialized yet", full_name());
134 return;
135 }
136 refs.emplace(*root);
137 std::stringstream ss {m_name};
138 for (std::string k {}; std::getline(ss, k, '.');) {
139 if (refs.top().isTable()) {
140 refs.emplace(refs.top()[k]);
141 } else {
142 log_error("Failed to register CONVAR '{}': Engine config root table is not a table", full_name());
143 return;
144 }
145 }
146 auto key = refs.top();
147 if (!key.isValid()) [[unlikely]] {
148 log_error("Failed to register CONVAR '{}': Engine config key value is not valid", full_name());
149 return;
150 }
151 if constexpr (!std::is_same_v<T, bool> && (std::is_integral_v<T> || std::is_floating_point_v<T>)) {
152 if (!key.isNumber()) {
153 log_error("Failed to register CONVAR '{}': Engine config key value is not of type 'number': '{}'", full_name(), key.tostring());
154 return;
155 }
156 } else if constexpr (std::is_same_v<T, bool>) {
157 if (!key.isBool()) {
158 log_error("Failed to register CONVAR '{}': Engine config key value is not of type 'boolean': '{}'", full_name(), key.tostring());
159 return;
160 }
161 } else if constexpr (std::is_same_v<T, std::string>) {
162 if (!key.isString()) {
163 log_error("Failed to register CONVAR '{}': Engine config key value is not of type 'string': '{}'", full_name(), key.tostring());
164 return;
165 }
166 }
167 m_ref = key;
168 detail::s_convars.emplace_back(std::make_pair(&m_ref, &m_is_locked));
169 }

171 [[nodiscard]] auto fallback() const -> T {
172 if (std::holds_alternative<T>(*m_fallback)) {
173 return std::get<T>(*m_fallback);
174 } else {
175 return std::get<std::function<auto()->T>>(*m_fallback)();
176 }
177 }

179 static constexpr auto type_name() noexcept -> std::string_view {
180 if constexpr (!std::is_same_v<T, bool> && (std::is_integral_v<T> || std::is_floating_point_v<T>)) return "number";
181 else if constexpr (std::is_same_v<T, bool>) return "boolean";
182 else if constexpr (std::is_same_v<T, std::string>) return "string";
183 else return "unknown";
184 }
