37void AttachableCollection<T>::Attach(Noesis::DependencyObject* associatedObject)
38{
39 if (mAssociatedObject != associatedObject)
40 {
41 if (mAssociatedObject != 0)
42 {
43 NS_ERROR("'%s' already attached to another object '%s'",
44 GetClassType()->GetName(), mAssociatedObject->GetClassType()->GetName());
45 return;
46 }
47
48 NS_ASSERT(associatedObject != 0);
49 mAssociatedObject = associatedObject;
50 mAssociatedObject->Destroyed() += MakeDelegate(this,
51 &AttachableCollection::OnAttachedDestroyed);
52
53 InitComponent(this, true);
54
55 OnAttached();
56 }
