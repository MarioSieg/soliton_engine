26 // An includer for files implementing shaderc's includer interface. It responds
27 // to the file including query from the compiler with the full path and content
28 // of the file to be included. In the case that the file is not found or cannot
29 // be opened, the full path field of in the response will point to an empty
30 // string, and error message will be passed to the content field.
31 // This class provides the basic thread-safety guarantee.
32 class FileIncluder : public shaderc::CompileOptions::IncluderInterface {
33 public:
34 explicit FileIncluder(const shaderc_util::FileFinder* file_finder)
35 : file_finder_(*file_finder) {}
36
37 ~FileIncluder() override;
38
39 // Resolves a requested source file of a given type from a requesting
40 // source into a shaderc_include_result whose contents will remain valid
41 // until it's released.
42 shaderc_include_result* GetInclude(const char* requested_source,
43 shaderc_include_type type,
44 const char* requesting_source,
45 size_t include_depth) override;
46 // Releases an include result.
47 void ReleaseInclude(shaderc_include_result* include_result) override;
48
49 // Returns a reference to the member storing the set of included files.
50 const ankerl::unordered_dense::set<eastl::string> & file_path_trace() const {
51 return included_files_;
52 }
53
54 private:
55 // Used by GetInclude() to get the full filepath.
56 const shaderc_util::FileFinder& file_finder_;
57 // The full path and content of a source file.
58 struct FileInfo {
59 const eastl::string full_path;
60 std::vector<char> contents;
61 };
62
63 // The set of full paths of included files.
64 ankerl::unordered_dense::set<eastl::string> included_files_;
65 };
