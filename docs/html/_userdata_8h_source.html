64 static Userdata* getClass(lua_State* L,
65 int index,
66 const void* registryConstKey,
67 const void* registryClassKey,
68 bool canBeConst)
69 {
70 const int result = lua_getmetatable(L, index); // Stack: object metatable (ot) | nil
71 if (result == 0 || !lua_istable(L, -1))
72 {
73 lua_rawgetp(L, LUA_REGISTRYINDEX, registryClassKey); // Stack: ot | nil, registry metatable (rt) | nil
74 return throwBadArg(L, index);
75 }
76
77 lua_rawgetp(L, -1, getConstKey()); // Stack: ot | nil, const table (co) | nil
78 LUABRIDGE_ASSERT(lua_istable(L, -1) || lua_isnil(L, -1));
79
80 // If const table is NOT present, object is const. Use non-const registry table
81 // if object cannot be const, so constness validation is done automatically.
82 // E.g. nonConstFn (constObj)
83 // -> canBeConst = false, isConst = true
84 // -> 'Class' registry table, 'const Class' object table
85 // -> 'expected Class, got const Class'
86 const bool isConst = lua_isnil(L, -1); // Stack: ot | nil, nil, rt
87 lua_rawgetp(L, LUA_REGISTRYINDEX, (isConst && canBeConst)
88 ? registryConstKey
89 : registryClassKey); // Stack: ot, co |Â nil, rt
90
91 lua_insert(L, -3); // Stack: rt, ot, co | nil
92 lua_pop(L, 1); // Stack: rt, ot
93
94 for (;;)
95 {
96 if (lua_rawequal(L, -1, -2)) // Stack: rt, ot
97 {
98 lua_pop(L, 2); // Stack: -
99 return static_cast<Userdata*>(lua_touserdata(L, index));
100 }
101
102 // Replace current metatable with it's base class.
103 lua_rawgetp(L, -1, getParentKey()); // Stack: rt, ot, parent ot (pot) | nil
104
105 if (lua_isnil(L, -1)) // Stack: rt, ot, nil
106 {
107 // Drop the object metatable because it may be some parent metatable
108 lua_pop(L, 2); // Stack: rt
109 return throwBadArg(L, index);
110 }
111
112 lua_remove(L, -2); // Stack: rt, pot
113 }
114
115 // no return
116 }
117
118 static bool isInstance(lua_State* L, int index, const void* registryKey)
119 {
120 const auto result = lua_getmetatable(L, index); // Stack: object metatable (ot) | nil
121 if (result == 0)
122 return false;
123
124 if (!lua_istable(L, -1))
125 {
126 lua_pop(L, 1); // Stack: -
127 return false;
128 }
129
130 lua_rawgetp(L, LUA_REGISTRYINDEX, registryKey); // Stack: ot, rt
131 lua_insert(L, -2); // Stack: rt, ot
132
133 for (;;)
134 {
135 if (lua_rawequal(L, -1, -2)) // Stack: rt, ot
136 {
137 lua_pop(L, 2); // Stack: -
138 return true;
139 }
140
141 // Replace current metatable with it's base class.
142 lua_rawgetp(L, -1, getParentKey()); // Stack: rt, ot, parent ot (pot) | nil
143
144 if (lua_isnil(L, -1)) // Stack: rt, ot, nil
145 {
146 // Drop the object metatable because it may be some parent metatable
147 lua_pop(L, 3); // Stack: -
148 return false;
149 }
150
151 lua_remove(L, -2); // Stack: rt, pot
152 }
153
154 // no return
155 }
156
157 static Userdata* throwBadArg(lua_State* L, int index)
158 {
159 LUABRIDGE_ASSERT(lua_istable(L, -1) || lua_isnil(L, -1)); // Stack: rt | nil
160
161 const char* expected = 0;
162 if (lua_isnil(L, -1)) // Stack: nil
163 {
164 expected = "unregistered class";
165 }
166 else
167 {
168 lua_rawgetp(L, -1, getTypeKey()); // Stack: rt, registry type
169 expected = lua_tostring(L, -1);
170 lua_pop(L, 1); // Stack: rt
171 }
172
173 const char* got = nullptr;
174 if (lua_isuserdata(L, index))
175 {
176 lua_getmetatable(L, index); // Stack: rt, ot | nil
177 if (lua_istable(L, -1)) // Stack: rt, ot
178 {
179 lua_rawgetp(L, -1, getTypeKey()); // Stack: rt, ot, object type | nil
180 if (lua_isstring(L, -1))
181 got = lua_tostring(L, -1);
182
183 lua_pop(L, 1); // Stack: rt, ot
184 }
185
186 lua_pop(L, 1); // Stack: rt
187 }
188
189 if (!got)
190 {
191 lua_pop(L, 1); // Stack
192 got = lua_typename(L, lua_type(L, index));
193 }
194
195 luaL_argerror(L, index, lua_pushfstring(L, "%s expected, got %s", expected, got));
196 return nullptr;
197 }
198
199public:
200 virtual ~Userdata() {}
201

235 template <class T>
236 static T* get(lua_State* L, int index, bool canBeConst)
237 {
238 if (lua_isnil(L, index))
239 return nullptr;
240
241 auto* clazz = getClass(L, index, detail::getConstRegistryKey<T>(), detail::getClassRegistryKey<T>(), canBeConst);
242 if (! clazz)
243 return nullptr;
244
245 return static_cast<T*>(clazz->getPointer());
246 }
247
248 template <class T>
249 static bool isInstance(lua_State* L, int index)
250 {
251 return isInstance(L, index, detail::getClassRegistryKey<T>())
252 || isInstance(L, index, detail::getConstRegistryKey<T>());
253 }
254
255protected:
256 Userdata() = default;
257
261 void* getPointer() const noexcept
262 {
263 return m_p;
264 }

266 void* m_p = nullptr; // subclasses must set this
267};

275template <class T>
276class UserdataValue : public Userdata
277{
278 using AlignType = typename std::conditional_t<alignof(T) <= alignof(double), T, void*>;
279
280 static constexpr int MaxPadding = alignof(T) <= alignof(AlignType) ? 0 : alignof(T) - alignof(AlignType) + 1;
281
282public:
283 UserdataValue(const UserdataValue&) = delete;
284 UserdataValue operator=(const UserdataValue&) = delete;
285
286 ~UserdataValue()
287 {
288 if (getPointer() != nullptr)
289 {
290 getObject()->~T();
291 }
292 }
293
303 static UserdataValue<T>* place(lua_State* L, std::error_code& ec)
304 {
305 auto* ud = new (lua_newuserdata_x<UserdataValue<T>>(L, sizeof(UserdataValue<T>))) UserdataValue<T>();
306
307 lua_rawgetp(L, LUA_REGISTRYINDEX, detail::getClassRegistryKey<T>());
308
309 if (!lua_istable(L, -1))
310 {
311 lua_pop(L, 1); // possibly: a nil
312
313 ud->~UserdataValue<T>();
314
315#if LUABRIDGE_RAISE_UNREGISTERED_CLASS_USAGE
316 ec = throw_or_error_code<LuaException>(L, ErrorCode::ClassNotRegistered);
317#else
318 ec = makeErrorCode(ErrorCode::ClassNotRegistered);
319#endif
320
321 return nullptr;
322 }
323
324 lua_setmetatable(L, -2);
325
326 return ud;
327 }
328
338 template <class U>
339 static auto push(lua_State* L, const U& u) -> std::enable_if_t<std::is_copy_constructible_v<U>, Result>
340 {
341 std::error_code ec;
342 auto* ud = place(L, ec);
343
344 if (!ud)
345 return ec;
346
347 new (ud->getObject()) U(u);
348
349 ud->commit();
350
351 return {};
352 }
353
363 template <class U>
364 static auto push(lua_State* L, U&& u) -> std::enable_if_t<std::is_move_constructible_v<U>, Result>
365 {
366 std::error_code ec;
367 auto* ud = place(L, ec);
368
369 if (!ud)
370 return ec;
371
372 new (ud->getObject()) U(std::move(u));
373
374 ud->commit();
375
376 return {};
377 }
378
382 void commit() noexcept
383 {
384 m_p = getObject();
385 }
386
387 T* getObject() noexcept
388 {
389 // If this fails to compile it means you forgot to provide
390 // a Container specialization for your container!
391
392 if constexpr (MaxPadding == 0)
393 return reinterpret_cast<T*>(&m_storage[0]);
394 else
395 return reinterpret_cast<T*>(&m_storage[0] + m_storage[sizeof(m_storage) - 1]);
396 }
397
398private:
402 UserdataValue() noexcept
403 : Userdata()
404 {
405 if constexpr (MaxPadding > 0)
406 {
407 uintptr_t offset = reinterpret_cast<uintptr_t>(&m_storage[0]) % alignof(T);
408 if (offset > 0)
409 offset = alignof(T) - offset;
410
411 assert(offset < MaxPadding);
412 m_storage[sizeof(m_storage) - 1] = static_cast<unsigned char>(offset);
413 }
414 }
415
416 alignas(AlignType) unsigned char m_storage[sizeof(T) + MaxPadding];
417};
