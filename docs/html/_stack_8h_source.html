34class StackRestore final
35{
36public:
37 StackRestore(lua_State* L)
38 : m_L(L)
39 , m_stackTop(lua_gettop(L))
40 {
41 }
42
43 ~StackRestore()
44 {
45 if (m_doRestoreStack)
46 lua_settop(m_L, m_stackTop);
47 }
48
49 void reset()
50 {
51 m_doRestoreStack = false;
52 }
53
54private:
55 lua_State* const m_L = nullptr;
56 int m_stackTop = 0;
57 bool m_doRestoreStack = true;
58};

232template <>
233struct Stack<char>
234{
235 [[nodiscard]] static Result push(lua_State* L, char value)
236 {
237#if LUABRIDGE_SAFE_STACK_CHECKS
238 if (! lua_checkstack(L, 1))
239 return makeErrorCode(ErrorCode::LuaStackOverflow);
240#endif
241
242 lua_pushlstring(L, &value, 1);
243 return {};
244 }
245
246 [[nodiscard]] static TypeResult<char> get(lua_State* L, int index)
247 {
248 if (lua_type(L, index) != LUA_TSTRING)
249 return makeErrorCode(ErrorCode::InvalidTypeCast);
250
251 std::size_t length = 0;
252 const char* str = lua_tolstring(L, index, &length);
253
254 if (str == nullptr || length != 1)
255 return makeErrorCode(ErrorCode::InvalidTypeCast);
256
257 return str[0];
258 }
259
260 [[nodiscard]] static bool isInstance(lua_State* L, int index)
261 {
262 if (lua_type(L, index) == LUA_TSTRING)
263 {
264 std::size_t len;
265 luaL_checklstring(L, index, &len);
266 return len == 1;
267 }
268
269 return false;
270 }
271};

1090#endif

1098 return result1;
1099

1101 if (! result2)
1102 return result2;
1103
1104 stackRestore.reset();
1105 return {};
1106 }

1109 {

1113 return makeErrorCode(ErrorCode::InvalidTypeCast);

1116 return makeErrorCode(ErrorCode::InvalidTableSizeInCast);

1118 std::pair<T1, T2> value;
1119

1121 lua_pushnil(L);
1122
1123 auto result1 = pop_element<0>(L, absIndex, value);
1124 if (! result1)
1125 return result1.error();
1126
1127 auto result2 = pop_element<1>(L, absIndex, value);
1128 if (! result2)
1129 return result2.error();
