38 [[nodiscard]] auto start_command_buffer() const -> vk::CommandBuffer {
39 vk::CommandBuffer cmd {};
40 vk::CommandBufferAllocateInfo cmd_alloc_info {};
41 vk::CommandPool pool {};
42 if constexpr (QueueType == vk::QueueFlagBits::eGraphics) {
43 pool = get_graphics_command_pool();
44 } else if constexpr (QueueType == vk::QueueFlagBits::eCompute) {
45 pool = get_compute_command_pool();
46 } else if constexpr (QueueType == vk::QueueFlagBits::eTransfer) {
47 pool = get_transfer_command_pool();
48 } else {
49 panic("Invalid queue type");
50 }
51 cmd_alloc_info.commandPool = pool;
52 cmd_alloc_info.level = vk::CommandBufferLevel::ePrimary;
53 cmd_alloc_info.commandBufferCount = 1;
54 vkcheck(m_device->get_logical_device().allocateCommandBuffers(&cmd_alloc_info, &cmd)); // TODO: not thread safe
55
56 vk::CommandBufferBeginInfo cmd_begin_info {};
57 cmd_begin_info.flags = vk::CommandBufferUsageFlagBits::eOneTimeSubmit;
58 vkcheck(cmd.begin(&cmd_begin_info));
59 return cmd;
60 }

62 auto flush_command_buffer(const vk::CommandBuffer cmd) const -> void {
63 const vk::Device device = m_device->get_logical_device();
64 if constexpr (Owned) {
65 vkcheck(cmd.end());
66 }
67 constexpr vk::FenceCreateInfo fence_info {};
68 vk::Fence fence {};
69 vkcheck(device.createFence(&fence_info, &vkb::s_allocator, &fence));
70 vk::SubmitInfo submit_info {};
71 submit_info.commandBufferCount = 1;
72 submit_info.pCommandBuffers = &cmd;
73 vk::Queue queue {};
74 if constexpr (QueueType == vk::QueueFlagBits::eGraphics) {
75 queue = m_device->get_graphics_queue();
76 } else if constexpr (QueueType == vk::QueueFlagBits::eCompute) {
77 queue = m_device->get_compute_queue();
78 } else if constexpr (QueueType == vk::QueueFlagBits::eTransfer) {
79 queue = m_device->get_transfer_queue();
80 } else {
81 panic("Invalid queue type");
82 }
83 vkcheck(queue.submit(1, &submit_info, fence));// TODO: not thread safe, use transfer queue
84 vkcheck(device.waitForFences(1, &fence, vk::True, std::numeric_limits<std::uint64_t>::max()));
85 device.destroyFence(fence, &vkb::s_allocator);
86 if constexpr (Owned) {
87 vk::CommandPool pool {};
88 if constexpr (QueueType == vk::QueueFlagBits::eGraphics) {
89 pool = get_graphics_command_pool();
90 } else if constexpr (QueueType == vk::QueueFlagBits::eCompute) {
91 pool = get_compute_command_pool();
92 } else if constexpr (QueueType == vk::QueueFlagBits::eTransfer) {
93 pool = get_transfer_command_pool();
94 } else {
95 panic("Invalid queue type");
96 }
97 device.freeCommandBuffers(pool, 1, &cmd);
98 }
