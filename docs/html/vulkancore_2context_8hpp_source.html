42 [[nodiscard]] auto start_command_buffer() const -> vk::CommandBuffer {
43 vk::CommandBuffer cmd {};
44 vk::CommandBufferAllocateInfo cmd_alloc_info {};
45 vk::CommandPool pool {};
46 if constexpr (QueueType == vk::QueueFlagBits::eGraphics) {
47 pool = get_graphics_command_pool();
48 } else if constexpr (QueueType == vk::QueueFlagBits::eCompute) {
49 pool = get_compute_command_pool();
50 } else if constexpr (QueueType == vk::QueueFlagBits::eTransfer) {
51 pool = get_transfer_command_pool();
52 } else {
53 panic("Invalid queue type");
54 }
55 cmd_alloc_info.commandPool = pool;
56 cmd_alloc_info.level = vk::CommandBufferLevel::ePrimary;
57 cmd_alloc_info.commandBufferCount = 1;
58 vkcheck(m_device->get_logical_device().allocateCommandBuffers(&cmd_alloc_info, &cmd)); // TODO: not thread safe
59
60 vk::CommandBufferBeginInfo cmd_begin_info {};
61 cmd_begin_info.flags = vk::CommandBufferUsageFlagBits::eOneTimeSubmit;
62 vkcheck(cmd.begin(&cmd_begin_info));
63 return cmd;
64 }

67 auto flush_command_buffer(const vk::CommandBuffer cmd) const -> void {
68 const vk::Device device = m_device->get_logical_device();
69 if constexpr (Owned) {
70 vkcheck(cmd.end());
71 }
72 constexpr vk::FenceCreateInfo fence_info {};
73 vk::Fence fence {};
74 vkcheck(device.createFence(&fence_info, vkb::get_alloc(), &fence));
75 vk::SubmitInfo submit_info {};
76 submit_info.commandBufferCount = 1;
77 submit_info.pCommandBuffers = &cmd;
78 vk::Queue queue {};
79 if constexpr (QueueType == vk::QueueFlagBits::eGraphics) {
80 queue = m_device->get_graphics_queue();
81 } else if constexpr (QueueType == vk::QueueFlagBits::eCompute) {
82 queue = m_device->get_compute_queue();
83 } else if constexpr (QueueType == vk::QueueFlagBits::eTransfer) {
84 queue = m_device->get_transfer_queue();
85 } else {
86 panic("Invalid queue type");
87 }
88 vkcheck(queue.submit(1, &submit_info, fence));// TODO: not thread safe, use transfer queue
89 vkcheck(device.waitForFences(1, &fence, vk::True, eastl::numeric_limits<std::uint64_t>::max()));
90 device.destroyFence(fence, vkb::get_alloc());
91 if constexpr (Owned) {
92 vk::CommandPool pool {};
93 if constexpr (QueueType == vk::QueueFlagBits::eGraphics) {
94 pool = get_graphics_command_pool();
95 } else if constexpr (QueueType == vk::QueueFlagBits::eCompute) {
96 pool = get_compute_command_pool();
97 } else if constexpr (QueueType == vk::QueueFlagBits::eTransfer) {
98 pool = get_transfer_command_pool();
99 } else {
100 panic("Invalid queue type");
101 }
102 device.freeCommandBuffers(pool, 1, &cmd);
103 }
104 }
