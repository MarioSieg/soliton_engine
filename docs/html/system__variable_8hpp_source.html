41 // System variable class. Configurable at runtime via INI or command line.
42 // Supported types: bool, float, int64_t, string.
43 template <typename T> requires sv_supported_type<T>
44 class system_variable final {
45 public:
46 constexpr system_variable(
47 eastl::string&& name,
48 eastl::variant<eastl::monostate, T, eastl::function<auto() -> T>>&& fallback
49 ) : m_name{name}, m_fallback{std::move(fallback)} {}
50
51 [[nodiscard]] auto operator()() const -> T {
52 std::lock_guard lock {detail::sv_mutex};
53 auto& reg = detail::sv_registry;
54 if (reg.contains(m_name)) [[likely]] { // If the variable is already registered, return it.
55 detail::value& val = reg[m_name];
56 if (eastl::holds_alternative<T>(val)) [[likely]] {
57 return eastl::get<T>(val);
58 } else {
59 panic("System variable '{}' is not of type '{}'", m_name, sv_supported_type_name<T>::value);
60 }
61 }
62 reg[m_name] = get_fallback(); // Otherwise, set the fallback value and return it
63 return eastl::get<T>(reg[m_name]);
64 }
65
66 private:
67 [[nodiscard]] auto get_fallback() const -> T {
68 if (eastl::holds_alternative<T>(m_fallback)) [[likely]]
69 return eastl::get<T>(m_fallback);
70 if (eastl::holds_alternative<eastl::function<auto() -> T>>(m_fallback))
71 return eastl::invoke(eastl::get<eastl::function<auto() -> T>>(m_fallback));
72 return T {};
73 }
74
75 const eastl::string m_name;
76 const eastl::variant<eastl::monostate, T, eastl::function<auto() -> T>> m_fallback;
77 };
