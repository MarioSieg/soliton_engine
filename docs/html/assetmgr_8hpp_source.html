58 [[nodiscard]] auto load_from_memory(Args&&... args) -> T* {
59 std::unique_ptr<T> ptr {std::make_unique<T>(std::forward<Args>(args)...)};
60 ++m_cache_misses;
61 return &*m_registry.emplace(fmt::format("mem_{:#X}", ++m_id_gen), std::move(ptr)).first->second;
62 }
63
64 template <typename... Args>
65 [[nodiscard]] auto load(std::string&& path, Args&&... args) -> T* {
66 if (m_registry.contains(path)) {
67 ++m_cache_hits;
68 return &*m_registry[path];
69 }
70 std::string key {path};
71 std::unique_ptr<T> ptr {std::make_unique<T>(std::move(path), std::forward<Args>(args)...)};
72 ++m_cache_misses;
73 return &*m_registry.emplace(std::move(key), std::move(ptr)).first->second;
74 }

82 [[nodiscard]] auto get_load_count() const noexcept -> std::uint32_t { return m_cache_hits + m_cache_misses; }
