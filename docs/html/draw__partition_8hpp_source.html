8 [[nodiscard]] inline auto compute_render_bucket_range(
9 const std::size_t id,
10 const std::size_t num_entities,
11 const std::size_t num_threads
12 ) noexcept -> eastl::array<std::size_t, 2> {
13 const std::size_t base_bucket_size = num_entities / num_threads;
14 const std::size_t num_extra_entities = num_entities % num_threads;
15 const std::size_t begin = base_bucket_size*id + eastl::min(id, num_extra_entities);
16 const std::size_t end = begin + base_bucket_size + (id < num_extra_entities ? 1 : 0);
17 panic_assert(begin <= end && end <= num_entities);
18 return {begin, end};
19 }

22 inline auto partitioned_draw(
23 const std::int32_t bucket_id,
24 const std::int32_t num_threads,
25 const eastl::vector<eastl::pair<eastl::span<const com::transform>, eastl::span<const com::mesh_renderer>>>& render_data,
26 F&& callback
27 ) -> void {
28 static constexpr auto accumulator = [](const std::size_t acc, const auto& pair) noexcept {
29 panic_assert(pair.first.size() == pair.second.size());
30 return acc + pair.first.size();
31 };
32 const std::size_t total_entities = std::accumulate(render_data.begin(), render_data.end(), 0, accumulator);
33 // Compute start and end for this thread across all entities
34 const auto [global_start, global_end] = compute_render_bucket_range(bucket_id, total_entities, num_threads);
35 std::size_t processed_entities = 0;
36 for (const auto& [transforms, renderers] : render_data) {
37 if (processed_entities >= global_end) break; // Already processed all entities this thread is responsible for
38 std::size_t local_start = 0;
39 std::size_t local_end = transforms.size();
40 if (processed_entities < global_start)
41 local_start = std::min(transforms.size(), global_start - processed_entities);
42 if (processed_entities + transforms.size() > global_end)
43 local_end = std::min(transforms.size(), global_end - processed_entities);
44 for (std::size_t i = local_start; i < local_end; ++i) {
45 eastl::invoke(callback, i, transforms[i], renderers[i]);
46 }
47 processed_entities += transforms.size();
48 }
49 }
