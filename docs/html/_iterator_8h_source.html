23public:

78 }

96 }
97 }
98
104 bool isNil() const noexcept
105 {
106 return m_key.isNil();
107 }
108

117 }

125 {
126 return m_value;
127 }
128
129private:
130 // Don't use postfix increment, it is less efficient
131 Iterator operator++(int);
132
133 void next()
134 {
135#if LUABRIDGE_SAFE_STACK_CHECKS
136 if (! lua_checkstack(m_L, 2))
137 {
138 m_key = LuaNil();
139 m_value = LuaNil();
140 return;
141 }
142#endif
143
144 m_table.push();
145 m_key.push();
146
147 if (lua_next(m_L, -2))
148 {
149 m_value.pop();
150 m_key.pop();
151 }
152 else
153 {
154 m_key = LuaNil();
155 m_value = LuaNil();
156 }
157
158 lua_pop(m_L, 1);
159 }
160
161 lua_State* m_L = nullptr;
162 LuaRef m_table;
163 LuaRef m_key;
164 LuaRef m_value;
165};
